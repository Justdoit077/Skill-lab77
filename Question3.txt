# Define variables x and y and assign integer values
x = 5
y = 10

# Print the values of x and y
print("Value of x:", x)
print("Value of y:", y)

# Perform logical operations: AND, OR, NOT
logical_and = x and y
logical_or = x or y
logical_not_x = not x
logical_not_y = not y

# Display the result of each operation
print("Logical AND (x and y):", logical_and)
print("Logical OR (x or y):", logical_or)
print("Logical NOT (not x):", logical_not_x)
print("Logical NOT (not y):", logical_not_y)
